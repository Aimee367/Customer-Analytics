Chi-sq Test

In[50]:
# Chi Sq Test
categorical_features = ['Contract', 'Payment Method', 'Internet Service']

for feature in categorical_features:
    table = pd.crosstab(telco_data[feature], telco_data['Churn Value'])
    chi2, p, _, _ = chi2_contingency(table)
    print(f"{feature} vs Churn - Chi-Square Statistic: {chi2}, p-value: {p}")
